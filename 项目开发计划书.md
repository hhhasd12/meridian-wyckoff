```python
class CandlePhysical:
    def __init__(self, open, high, low, close, volume):
        self.open = open
        self.high = high
        self.low = low
        self.close = close
        self.volume = volume
        
    @property
    def body(self):
        """实体（控制力度）"""
        return abs(self.close - self.open)
    
    @property
    def upper_shadow(self):
        """上影线长度"""
        return self.high - max(self.open, self.close)
    
    @property
    def lower_shadow(self):
        """下影线长度"""
        return min(self.open, self.close) - self.low
    
    @property
    def total_shadow(self):
        """总影线长度（针的极限力度）"""
        return self.upper_shadow + self.lower_shadow
    
    @property
    def body_ratio(self):
        """实体占比 = 实体 / (实体+影线)"""
        total_range = self.high - self.low
        return self.body / total_range if total_range > 0 else 0
    
    @property
    def shadow_ratio(self):
        """影线占比 = 影线 / (实体+影线)"""
        total_range = self.high - self.low
        return self.total_shadow / total_range if total_range > 0 else 0
    
    def get_intensity_score(self):
        """强度评分 = 影线占比 * 成交量系数"""
        shadow_importance = self.shadow_ratio * 2.0
        volume_factor = self.volume / volume_moving_avg
        return shadow_importance * min(volume_factor, 3.0)
```

### 2.2 "针vs实体"辩证识别算法
```python
def analyze_pin_vs_body(candle, context):
    """
    分析单根K线的针与实体关系，使用动态参数系统
    返回：{'is_pin_dominant': bool, 'is_body_dominant': bool, 'confidence': float}
    """
    results = {
        'is_pin_dominant': False,
        'is_body_dominant': False,
        'pin_strength': 0.0,
        'body_strength': 0.0,
        'effort_vs_result': None,
        'dynamic_thresholds_used': {}  # 记录使用的动态阈值
    }
    
    body_size = candle.body
    shadow_size = candle.total_shadow
    total_range = candle.high - candle.low
    
    # ===== 动态参数计算 =====
    # 1. 波动率因子：当前ATR相对于历史平均ATR的比例
    volatility_factor = context.volatility_index  # current_atr / avg_atr
    # 示例：低波动时0.8，正常时1.0，高波动时1.5
    
    # 2. 市场体制因子：根据市场状态调整阈值
    regime_factor = context.market_regime_factor
    # 示例：趋势市1.2（实体更重要），盘整市0.9（针更重要）
    
    # 3. 动态阈值计算
    # 基础阈值（正常波动率下的经验值）
    BASE_PIN_THRESHOLD = 1.5      # 针主导的基础阈值
    BASE_BODY_THRESHOLD = 2.0     # 实体主导的基础阈值  
    BASE_VOLUME_SPIKE = 1.8       # 成交量爆发阈值
    BASE_EFFORT_THRESHOLD = 2.0   # 高努力阈值
    BASE_RESULT_THRESHOLD = 0.5   # 低结果阈值
    
    # 动态调整：高波动时需要更大的针/实体比例
    dynamic_pin_threshold = BASE_PIN_THRESHOLD * volatility_factor
    dynamic_body_threshold = BASE_BODY_THRESHOLD * volatility_factor
    
    # 市场体制调整：趋势市实体阈值降低，盘整市针阈值降低
    if context.market_regime == 'TRENDING':
        dynamic_body_threshold *= 0.9  # 趋势中实体更容易出现
        dynamic_pin_threshold *= 1.1   # 趋势中针更少见
    elif context.market_regime == 'RANGING':
        dynamic_pin_threshold *= 0.9   # 盘整中针更常见
        dynamic_body_threshold *= 1.1  # 盘整中实体更少见
    
    # 记录使用的动态阈值
    results['dynamic_thresholds_used'] = {
        'pin_threshold': dynamic_pin_threshold,
        'body_threshold': dynamic_body_threshold,
        'volatility_factor': volatility_factor,
        'regime_factor': regime_factor,
        'market_regime': context.market_regime
    }
    
    # ===== 针主导判定（使用动态阈值） =====
    if shadow_size > body_size * dynamic_pin_threshold:
        results['is_pin_dominant'] = True
        results['pin_strength'] = shadow_size / total_range
        
        # 成交量爆发阈值也动态调整
        dynamic_volume_spike = BASE_VOLUME_SPIKE / volatility_factor
        # 低波动时成交量爆发更显著（阈值降低），高波动时阈值提高
        if candle.volume > context.volume_moving_avg * dynamic_volume_spike:
            results['pin_strength'] *= 1.5
            results['volume_confirmation'] = True
    
    # ===== 实体主导判定（使用动态阈值） =====
    elif body_size > shadow_size * dynamic_body_threshold:
        results['is_body_dominant'] = True
        results['body_strength'] = body_size / total_range
        if is_at_support_resistance(candle, context):
            results['body_strength'] *= 1.3
    
    # ===== 努力与结果分析（使用动态阈值） =====
    effort = candle.volume
    result = body_size
    
    # 动态调整努力与结果阈值
    dynamic_effort_threshold = BASE_EFFORT_THRESHOLD / volatility_factor
    dynamic_result_threshold = BASE_RESULT_THRESHOLD * volatility_factor
    
    if (effort > context.volume_moving_avg * dynamic_effort_threshold and 
        result < context.avg_body_size * dynamic_result_threshold):
        results['effort_vs_result'] = 'HIGH_EFFORT_LOW_RESULT'
        results['effort_ratio'] = effort / context.volume_moving_avg
        results['result_ratio'] = result / context.avg_body_size
    
    elif (effort < context.volume_moving_avg * 0.7 and 
          result > context.avg_body_size * 1.5):
        results['effort_vs_result'] = 'LOW_EFFORT_HIGH_RESULT'
    
    return results
```

#### 2.4.1 基础TR识别算法
```python
class TradingRangeDetector:
    def __init__(self, config):
        self.config = config  # 动态参数配置
        self.locked_tr = None  # 已锁定的TR
        self.lock_counter = 0  # 锁定计数器
        self.breakout_confirmed = False  # 突破确认标志
    
    def identify_tr(self, price_bars, volume_bars):
        """
        智能识别交易区间，包含稳定性锁定机制
        返回: {
            'type': 'HORIZONTAL'/'ASCENDING'/'DESCENDING'/'NONE',
            'support': float,
            'resistance': float,
            'slope': float,
            'quality_score': float,
            'is_locked': bool,  # 是否处于锁定状态
            'breakout_attempts': int  # 突破尝试次数
        }
        """
        # 1. 基础TR识别（线性回归 + 价格包络线）
        base_tr = self._identify_base_tr(price_bars, volume_bars)
        
        # 2. 稳定性锁定检查
        if self.locked_tr and not self.breakout_confirmed:
            # 检查是否有效突破锁定区间
            breakout_status = self._check_breakout(base_tr, price_bars[-1])
            
            if breakout_status['confirmed']:
                # 确认突破 → 解除锁定
                self.locked_tr = None
                self.lock_counter = 0
                self.breakout_confirmed = True
            else:
                # 未确认突破 → 保持锁定
                return {
                    **self.locked_tr,
                    'is_locked': True,
                    'lock_reason': breakout_status['reason']
                }
        
        # 3. 新TR锁定条件
        if (base_tr['quality_score'] > self.config.TR_LOCK_MIN_QUALITY and
            base_tr['type'] != 'NONE' and
            self._is_tr_stable(base_tr, price_bars)):
            
            # 满足锁定条件 → 锁定该TR
            self.locked_tr = base_tr
            self.lock_counter = 0
            self.breakout_confirmed = False
            
            base_tr['is_locked'] = True
            base_tr['lock_timestamp'] = get_current_time()
        
        return {**base_tr, 'is_locked': False}
    
    def _check_breakout(self, locked_tr, current_price):
        """
        检查是否有效突破锁定区间
        规则：价格必须突破边界N%并保持M根K线
        """
        # 动态参数
        breakout_percent = self.config.TR_BREAKOUT_PERCENT  # 默认1.5%
        confirmation_bars = self.config.TR_CONFIRMATION_BARS  # 默认3根
        
        support = locked_tr['support']
        resistance = locked_tr['resistance']
        
        # 上突破检查
        if current_price.close > resistance * (1 + breakout_percent/100):
            self.lock_counter += 1
            if self.lock_counter >= confirmation_bars:
                return {
                    'confirmed': True,
                    'direction': 'UP',
                    'breakout_percent': (current_price.close - resistance) / resistance * 100
                }
        
        # 下突破检查
        elif current_price.close < support * (1 - breakout_percent/100):
            self.lock_counter += 1
            if self.lock_counter >= confirmation_bars:
                return {
                    'confirmed': True,
                    'direction': 'DOWN',
                    'breakout_percent': (support - current_price.close) / support * 100
                }
        else:
            # 价格回到区间内 → 重置计数器
            self.lock_counter = max(0, self.lock_counter - 1)
        
        return {
            'confirmed': False,
            'reason': f'需要{confirmation_bars-self.lock_counter}根K线确认突破'
        }
    
    def _is_tr_stable(self, tr, price_bars):
        """
        判断TR是否稳定（防止线性回归频繁跳动）
        稳定性条件：
        1. 斜率变化率 < 阈值
        2. 边界波动率 < 阈值
        3. 持续至少N根K线
        """
        if len(price_bars) < self.config.TR_MIN_BARS:
            return False
        
        # 计算斜率稳定性
        slope_stability = self._calculate_slope_stability(price_bars)
        
        # 计算边界稳定性
        boundary_stability = self._calculate_boundary_stability(price_bars, tr)
        
        return (slope_stability > self.config.SLOPE_STABILITY_THRESHOLD and
                boundary_stability > self.config.BOUNDARY_STABILITY_THRESHOLD)
```

#### 2.4.3 设计原则
1. **防抖动机制**：一旦TR被锁定，除非价格有效突破边界N%并保持M根K线，否则不改变斜率定义
2. **渐进式解锁**：突破尝试需要连续确认，单次假突破不会立即解除锁定
3. **动态适应性**：所有阈值均为动态参数，可随市场波动率自动调整
4. **质量优先**：只有高质量的TR才被锁定，避免锁定噪音区间

### 2.5 异常数据清洗模块（Data Sanitization）

#### 2.5.1 数据质量问题与清洗策略
VSA（量价分析）对数据质量要求极高，以下异常情况需要清洗：
1. **交易所停机维护**：零成交量或异常价格的K线
2. **数据源杂波**：价格跳空超过N倍ATR的异常波动
3. **成交量异常**：成交量超过历史均值的10倍以上
4. **时间戳错误**：K线时间戳不连续或重叠

```python
class AnomalyEvent:
    """
    异常事件对象 - 用于封装异常K线数据，而非清洗
    VSA分析的核心就是异常，不应平滑或插值异常数据
    """
    def __init__(self, raw_candle, anomaly_types, anomaly_score, market_type):
        self.raw_candle = raw_candle  # 原始K线数据（不修改）
        self.anomaly_types = anomaly_types  # 异常类型列表
        self.anomaly_score = anomaly_score  # 异常严重度分数 [0-1]
        self.market_type = market_type  # 市场类型：STOCK/CRYPTO/FOREX
        self.timestamp = raw_candle.timestamp
        self.impact_score = 0.0  # 市场影响评分（由状态机计算）
        self.main_intent = None  # 主力意图分析结果
        
        # 自动分类异常事件类型
        if 'EXTREME_GAP' in anomaly_types:
            self.event_category = 'PRICE_GAP_EVENT'
            self.suggested_action = 'ANALYZE_AS_BREAKOUT_OR_EXHAUSTION'
        elif 'ZERO_VOLUME' in anomaly_types:
            self.event_category = 'LIQUIDITY_EVENT'
            self.suggested_action = 'CHECK_EXCHANGE_STATUS'
        elif 'EXTREME_VOLUME' in anomaly_types:
            self.event_category = 'VOLUME_CLIMAX_EVENT'
            self.suggested_action = 'ANALYZE_AS_SC_OR_BC'
        else:
            self.event_category = 'GENERIC_ANOMALY'
            self.suggested_action = 'PASS_TO_STATE_MACHINE'
    
    def to_state_machine_input(self):
        """
        转换为状态机输入格式
        """
        return {
            'type': 'ANOMALY_EVENT',
            'event_category': self.event_category,
            'anomaly_score': self.anomaly_score,
            'raw_data': {
                'open': self.raw_candle.open,
                'high': self.raw_candle.high,
                'low': self.raw_candle.low,
                'close': self.raw_candle.close,
                'volume': self.raw_candle.volume,
                'timestamp': self.timestamp
            },
            'market_context': self.market_type,

class DataSanitizer:
    def __init__(self, config):
        self.config = config
        self.market_type = config.MARKET_TYPE  # STOCK/CRYPTO/FOREX
        self.circuit_breaker_active = False  # 熔断机制状态
        self.circuit_breaker_start_time = None
        self.anomaly_history = []  # 异常事件历史记录
        
    def sanitize_candle(self, raw_candle, historical_context):
        """
        检测K线异常，但不清洗数据
        返回: (data_object, is_anomaly, anomaly_event)
        - 正常数据: 返回原始K线
        - 异常数据: 返回AnomalyEvent对象，供状态机分析主力意图
        """
        # 1. 检查熔断机制（加密/外汇市场）
        if self._check_circuit_breaker(raw_candle, historical_context):
            return self._handle_circuit_breaker(raw_candle, historical_context)
        
        # 2. 异常检测
        checks = [
            self._check_volume_anomaly(raw_candle, historical_context),
            self._check_price_gap(raw_candle, historical_context),
            self._check_timestamp_consistency(raw_candle, historical_context),
            self._check_range_anomaly(raw_candle, historical_context)
        ]
        
        # 3. 计算异常分数
        anomaly_score = sum(check['score'] for check in checks)
        is_anomaly = anomaly_score > self.config.ANOMALY_THRESHOLD
        
        if is_anomaly:
            # 异常数据 → 创建AnomalyEvent对象（不修改原始数据）
            anomaly_types = [check['type'] for check in checks if check['score'] > 0]
            anomaly_event = AnomalyEvent(
                raw_candle=raw_candle,
                anomaly_types=anomaly_types,
                anomaly_score=anomaly_score,
                market_type=self.market_type
            )
            
            # 记录异常事件
            self.anomaly_history.append({
                'timestamp': raw_candle.timestamp,
                'event': anomaly_event,
                'score': anomaly_score
            })
            
            # 检查是否需要触发熔断（加密/外汇市场的数据中断）
            if self._should_trigger_circuit_breaker(anomaly_types, historical_context):
                self._activate_circuit_breaker(anomaly_event)
            
            return anomaly_event, True, anomaly_event
        
        # 4. 正常数据 → 直接返回原始K线
        return raw_candle, False, None
    
    def _check_circuit_breaker(self, candle, context):
        """
        检查熔断机制是否激活
        规则：加密/外汇市场出现连续N根零成交量K线时激活
        """
        if not self.circuit_breaker_active:
            return False
        
        # 检查熔断是否超时
        if self.circuit_breaker_start_time:
            time_since_activation = candle.timestamp - self.circuit_breaker_start_time
            max_duration = self.config.CIRCUIT_BREAKER_MAX_DURATION[self.market_type]
            if time_since_activation > max_duration:
                self._deactivate_circuit_breaker()
                return False
        
        return self.circuit_breaker_active
    
    def _should_trigger_circuit_breaker(self, anomaly_types, context):
        """
        判断是否应该触发熔断机制
        规则：加密/外汇市场出现数据中断时触发
        """
        if self.market_type == 'STOCK':
            return False  # 股票市场允许插值
        
        # 加密/外汇市场：零成交量或极端跳空且连续出现
        if ('ZERO_VOLUME' in anomaly_types or 'EXTREME_GAP' in anomaly_types):
            # 检查是否连续异常
            recent_anomalies = len([e for e in self.anomaly_history[-3:] 
                                   if e['score'] > 0.8])
            return recent_anomalies >= 2
        
        return False
    
    def _activate_circuit_breaker(self, anomaly_event):
        """
        激活熔断机制
        """
        self.circuit_breaker_active = True
        self.circuit_breaker_start_time = anomaly_event.timestamp
        
        print(f"⚠️ 熔断机制激活 | 市场: {self.market_type} | 原因: {anomaly_event.event_category}")
        print(f"   系统将停止交易，直到接收到至少 {self.config.CIRCUIT_BREAKER_RECOVERY_BARS} 根正常K线")
    
    def _deactivate_circuit_breaker(self):
        """
        解除熔断机制
        """
        self.circuit_breaker_active = False
        self.circuit_breaker_start_time = None
        print("✅ 熔断机制解除 | 恢复正常数据处理")
    
    def _handle_circuit_breaker(self, raw_candle, context):
        """
        熔断状态下的数据处理
        返回特殊标记，通知系统停止交易
        """
        circuit_event = AnomalyEvent(
            raw_candle=raw_candle,
            anomaly_types=['CIRCUIT_BREAKER_ACTIVE'],
            anomaly_score=1.0,
            market_type=self.market_type
        )
        circuit_event.event_category = 'CIRCUIT_BREAKER_EVENT'
        circuit_event.suggested_action = 'STOP_TRADING_UNTIL_RECOVERY'
        
        return circuit_event, True, circuit_event
    
    def _check_volume_anomaly(self, candle, context):
        """
        检查成交量异常（VSA分析的核心就是异常，不应清洗）
        """
        if candle.volume == 0:
            return {'type': 'ZERO_VOLUME', 'score': 1.0}
        
        volume_ratio = candle.volume / context.volume_ma50
        if volume_ratio > self.config.MAX_VOLUME_RATIO:
            return {'type': 'EXTREME_VOLUME', 'score': 0.8}
        
        return {'type': 'NORMAL', 'score': 0.0}
    
    def _check_price_gap(self, candle, context):
        """
        检查价格跳空（异常就是信号，不应平滑）
        """
        if context.previous_close is None:
            return {'type': 'NORMAL', 'score': 0.0}
        
        gap_percent = abs(candle.open - context.previous_close) / context.previous_close
        atr_multiple = gap_percent / context.atr14
        
        if atr_multiple > self.config.MAX_GAP_ATR_MULTIPLE:
            return {'type': 'EXTREME_GAP', 'score': 0.9}
        
        return {'type': 'NORMAL', 'score': 0.0}
    
    # 其他检查方法保持不变（_check_timestamp_consistency, _check_range_anomaly）
    # 但删除所有插值和平滑处理方法
```

#### 2.5.4 设计原则
1. **异常即信号原则**：VSA分析的核心就是异常。禁止对价格跳空、极端成交量等异常数据进行平滑或插值处理。异常是主力资金留下的"脚印"，应作为信号分析而非噪声清洗。

2. **事件封装传递**：检测到异常时，将其封装为`Anomaly_Event`对象，包含原始数据、异常类型、严重度评分。事件对象直接输入状态机进行"主力意图分析"（例如：跳空不补=强势信号，跳空回补=假突破）。

3. **市场类型敏感处理**：
   - **股票市场**：允许有限的插值（如停牌导致的零成交量）
   - **加密/外汇市场**：禁止插值。数据中断（零成交量/极端跳空）99%是交易所API或网络问题，插值会生成虚假的"平稳K线"，网络恢复时价格可能已跳空20%

4. **熔断机制保护**：加密/外汇市场检测到数据中断时，触发系统熔断（Circuit Breaker），强制空仓或停止交易，直到接收到至少N根正常的实时K线。防止基于虚假数据的错误决策。

5. **可追溯分析**：记录所有异常事件、熔断激活/解除时间、状态机处理结果。便于事后分析异常数据的市场影响和主力意图。

```python
    def _calculate_state_intensity(self, state_name, detection_result):
        """
        计算状态强度
        SC强度 = f(成交量倍数, 针强度, 努力结果不匹配度)
        AR强度 = f(反弹幅度, 缩量程度)
        ST强度 = f(缩量程度, 价格回撤深度)
        """
        if state_name == 'SC':
            volume_ratio = detection_result.get('volume_ratio', 1.0)
            pin_strength = detection_result.get('pin_strength', 0.0)
            effort_result = 1.0 if detection_result.get('effort_vs_result') == 'HIGH_EFFORT_LOW_RESULT' else 0.0
            
            # SC强度公式：成交量权重40% + 针强度权重40% + 努力结果权重20%
            intensity = (volume_ratio * 0.4 + pin_strength * 0.4 + effort_result * 0.2)
            return min(intensity, 1.0)
        
        elif state_name == 'AR':
            bounce_percent = detection_result.get('bounce_percent', 0.0)
            volume_contraction = detection_result.get('volume_contraction', 1.0)
            
            # AR强度公式：反弹幅度权重60% + 缩量程度权重40%
            intensity = (min(bounce_percent, 0.1) * 10 * 0.6 +  # 10%反弹=1.0
                        (1.0 / volume_contraction) * 0.4)
            return min(intensity, 1.0)
        
        elif state_name == 'ST':
            volume_contraction = detection_result.get('volume_contraction', 1.0)
            retracement_depth = detection_result.get('retracement_depth', 0.0)
            
            # ST强度公式：缩量程度权重70% + 回撤深度权重30%
            # 极度缩量（<30%）得分高，适度缩量（60%）得分中等
            volume_score = max(0, (1.0 - volume_contraction) / 0.7)
            retracement_score = 1.0 - min(abs(retracement_depth - 0.5), 0.5) * 2  # 回撤50%最佳
            
            intensity = volume_score * 0.7 + retracement_score * 0.3
            return min(intensity, 1.0)
        
        # 其他状态默认强度计算
        return detection_result.get('confidence', 0.0)
    
    def _calculate_heritage_score(self, state_name):
        """
        计算从父状态继承的强度遗产
        规则：强SC → 强遗产 → ST需要极度缩量
        """
        if not self.heritage_chain:
            return 0.0
        
        current_state = self.accumulation_states[state_name]
        parent_states = current_state['parent_states']
        
        if not parent_states:
            return 0.0
        
        # 从遗产链中查找最近的父状态
        heritage_scores = []
        for heritage in reversed(self.heritage_chain):
            if heritage['state'] in parent_states:
                # 应用遗产传递规则
                parent_state = self.accumulation_states[heritage['state']]
                if 'heritage_rules' in parent_state:
                    rule_key = f'to_{state_name}'
                    if rule_key in parent_state['heritage_rules']:
                        transfer_func = parent_state['heritage_rules'][rule_key]
                        heritage_score = transfer_func(heritage['intensity'])
                        heritage_scores.append(heritage_score)
                else:
                    # 默认传递：父强度 * 衰减系数
                    heritage_scores.append(heritage['intensity'] * 0.7)
        
        if heritage_scores:
            return max(heritage_scores)
        
        return 0.0
    
    def get_state_requirements(self, state_name):
        """
        获取状态的动态要求（考虑遗产强度）
        例如：强遗产SC后，ST需要更严格的缩量要求
        """
        state = self.accumulation_states.get(state_name)
        if not state:
            return {}
        
        requirements = {}
        heritage_score = state['heritage_score']
        
        if state_name == 'ST':
            # ST的成交量要求随遗产强度变化
            if heritage_score > 0.8:
                requirements['max_volume_ratio'] = 0.3  # 强遗产：必须极度缩量（<30%）
            elif heritage_score > 0.5:
                requirements['max_volume_ratio'] = 0.5  # 中等遗产：适度缩量（<50%）
            else:
                requirements['max_volume_ratio'] = 0.7  # 弱遗产：普通缩量（<70%）
            
            # 价格回撤要求
            requirements['ideal_retracement'] = 0.5  # 理想回撤50%
            requirements['retracement_tolerance'] = 0.2 * (1.0 - heritage_score)  # 强遗产容忍度小
        
        elif state_name == 'AR':
            # AR的反弹要求
            requirements['min_bounce_percent'] = 0.02 + heritage_score * 0.03  # 1-5%
            requirements['max_volume_ratio'] = 0.8  # AR应缩量
        
        return requirements
```

### 3.2 关键节点检测算法（SC为例）
```python
def detect_sc(self, candle, context):
    """
    检测抛售高潮的辩证维度
    返回：{'is_sc': bool, 'confidence': float, 'evidence': dict}
    """
    evidence = {
        'volume_spike': False,      # 成交量爆发（2倍以上）
        'pin_dominant': False,      # 针主导（长下影线）
        'trend_context': False,     # 趋势背景（下跌末端）
        'position_in_tr': False,    # 交易区间位置（支撑位）
        'effort_result_mismatch': False,  # 努力与结果不匹配
        'follow_up_action': None    # 后续K线验证
    }
    
    confidence = 0.0
    required_confidence = 0.7
    
    # 1. 成交量爆发
    volume_ratio = candle.volume / context.volume_ma20
    if volume_ratio >= 2.0:
        evidence['volume_spike'] = True
        confidence += 0.25
    
    # 2. 针主导
    pin_analysis = analyze_pin_vs_body(candle, context)
    if pin_analysis['is_pin_dominant'] and pin_analysis['pin_strength'] > 0.6:
        evidence['pin_dominant'] = True
        confidence += 0.20
    
    # 3. 趋势背景
    if context.trend == 'DOWNTREND' and context.trend_strength < 0.3:
        evidence['trend_context'] = True
        confidence += 0.15
    
    # 4. 位置分析
    if context.tr_support and abs(candle.low - context.tr_support) / context.tr_support < 0.02:
        evidence['position_in_tr'] = True
        confidence += 0.10
    
    # 5. 努力与结果
    if pin_analysis['effort_vs_result'] == 'HIGH_EFFORT_LOW_RESULT':
        evidence['effort_result_mismatch'] = True
        confidence += 0.15
    
    # 6. 后续验证
    if len(context.recent_candles) >= 3:
        next_candle = context.recent_candles[-1]
        if next_candle.volume < candle.volume * 0.6 and next_candle.close > candle.close:
            evidence['follow_up_action'] = 'CONFIRMED'
            confidence += 0.15
        elif next_candle.volume > candle.volume and next_candle.close < candle.close:
            evidence['follow_up_action'] = 'REJECTED'
            confidence -= 0.30
    
    is_sc = confidence >= required_confidence and \
            sum([1 for v in evidence.values() if v is True]) >= 3
    
    return {
        'is_sc': is_sc,
        'confidence': min(max(confidence, 0.0), 1.0),
        'evidence': evidence
    }
```

### 3.3 非线性状态转换与重置机制

#### 3.3.1 状态转换核心原则
1. **非线性跳转支持**：允许跳过中间状态（如SC→直接拉升，跳过ST）
2. **状态重置触发**：关键价格突破时所有相关状态归零
3. **并行状态维护**：同时跟踪多个可能性路径
4. **超时强制退出**：防止状态机"死锁"

#### 3.3.2 状态机增强实现
```python
class EnhancedWyckoffStateMachine(WyckoffStateMachine):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.critical_price_levels = {}  # 关键价格水平
        self.alternative_paths = []      # 并行路径
        self.state_timeout_counters = {} # 状态超时计数器
        
    def process_candle_with_reset(self, candle, context):
        """
        处理K线，包含状态重置检查
        """
        # 1. 检查状态重置条件（最高优先级）
        reset_required = self._check_state_reset_conditions(candle, context)
        if reset_required:
            self._perform_state_reset(reset_required['reason'], reset_required['new_base_state'])
        
        # 2. 非线性状态检测（并行检测所有可能状态）
        potential_states = self._detect_nonlinear_states(candle, context)
        
        # 3. 更新并行路径
        self._update_alternative_paths(potential_states, candle)
        
        # 4. 选择最佳路径（基于证据权重）
        best_state = self._select_best_path()
        
        # 5. 执行状态转换
        if best_state and best_state != self.current_state:
            self._execute_state_transition(best_state, candle.timestamp)
        
        return self.current_state
    
    def _check_state_reset_conditions(self, candle, context):
        """
        检查状态重置条件
        返回: None 或 {'reason': str, 'new_base_state': str}
        """
        reset_conditions = [
            # 条件1: 价格跌破SC低点 → 所有吸筹状态归零
            {
                'condition': (self.current_state in ['SC', 'AR', 'ST', 'TEST'] and 
                             candle.close < self.critical_price_levels.get('SC_LOW', float('inf'))),
                'reason': 'PRICE_BREAKS_SC_LOW',
                'new_base_state': 'IDLE',
                'reset_scope': 'ACCUMULATION'  # 重置所有吸筹相关状态
            },
            # 条件2: 价格突破JOC高点 → 吸筹完成，转为趋势
            {
                'condition': (self.current_state in ['JOC', 'BU'] and
                             candle.close > self.critical_price_levels.get('JOC_HIGH', 0)),
                'reason': 'PRICE_CONFIRMS_JOC',
                'new_base_state': 'TREND_UP',
                'reset_scope': 'ACCUMULATION'  # 吸筹阶段结束
            },
            # 条件3: Spring失败（价格未弹回区间）→ 转为下跌
            {
                'condition': (self.current_state == 'SPRING' and
                             context.bars_since_spring > self.config.SPRING_FAILURE_BARS and
                             candle.close < self.critical_price_levels.get('SPRING_LOW', float('inf'))),
                'reason': 'SPRING_FAILURE',
                'new_base_state': 'DOWNTREND',
                'reset_scope': 'ACCUMULATION'
            },
            # 条件4: 超时强制重置
            {
                'condition': (self.current_state != 'IDLE' and
                             self.state_timeout_counters.get(self.current_state, 0) > 
                             self.config.STATE_TIMEOUT_BARS),
                'reason': 'STATE_TIMEOUT',
                'new_base_state': 'IDLE',
                'reset_scope': 'ALL'
            }
        ]
        
        for condition in reset_conditions:
            if condition['condition']:
                return {
                    'reason': condition['reason'],
                    'new_base_state': condition['new_base_state'],
                    'reset_scope': condition['reset_scope']
                }
        
        return None
    
    def _perform_state_reset(self, reason, new_base_state):
        """
        执行状态重置
        """
        print(f"状态重置: {reason} | {self.current_state} -> {new_base_state}")
        
        # 重置状态机
        self.current_state = new_base_state
        self.state_history.append({
            'timestamp': get_current_time(),
            'from_state': 'RESET',
            'to_state': new_base_state,
            'reason': reason,
            'type': 'RESET'
        })
        
        # 根据重置范围清理相关状态
        if reason == 'PRICE_BREAKS_SC_LOW':
            # 清除所有吸筹相关状态的置信度
            for state in self.accumulation_states:
                self.confidence_scores[state] = 0.0
            # 清除关键价格水平
            self.critical_price_levels.pop('SC_LOW', None)
        
        # 重置超时计数器
        self.state_timeout_counters = {}
    
    def _detect_nonlinear_states(self, candle, context):
        """
        非线性状态检测：允许从任何状态跳转到任何其他状态
        基于当前证据强度，而非历史路径
        """
        potential_states = []
        
        # 独立检测每个状态的可能性（不依赖前驱状态）
        for state_name, state_info in self.accumulation_states.items():
            detection_method = getattr(self, f'detect_{state_name.lower()}', None)
            if detection_method:
                result = detection_method(candle, context)
                if result['confidence'] > self.config.STATE_MIN_CONFIDENCE:
                    potential_states.append({
                        'state': state_name,
                        'confidence': result['confidence'],
                        'evidence': result['evidence'],
                        'direct_jump': True  # 标记为直接跳转
                    })
        
        return potential_states
    
    def _update_alternative_paths(self, potential_states, candle):
        """
        更新并行路径跟踪
        """
        # 添加新发现的路径
        for state_info in potential_states:
            # 检查是否已存在类似路径
            existing_path = next((p for p in self.alternative_paths 
                                 if p['state'] == state_info['state']), None)
            
            if existing_path:
                # 更新现有路径置信度
                existing_path['confidence'] = max(
                    existing_path['confidence'],
                    state_info['confidence']
                )
                existing_path['last_updated'] = candle.timestamp
            else:
                # 添加新路径
                self.alternative_paths.append({
                    **state_info,
                    'created_at': candle.timestamp,
                    'last_updated': candle.timestamp,
                    'age_bars': 0
                })
        
        # 清理过时路径（超过N根K线未更新）
        self.alternative_paths = [
            p for p in self.alternative_paths
            if p['age_bars'] < self.config.PATH_MAX_AGE_BARS
        ]
    
    def _select_best_path(self):
        """
        从并行路径中选择最佳路径，带有滞后性（Hysteresis）防止"精神分裂"
        规则：新路径必须显著优于当前路径才切换（例如>15%）
        """
        if not self.alternative_paths:
            return None
        
        # 计算当前状态的综合评分（如果当前状态不在并行路径中，使用历史记录）
        current_state_score = self._get_current_state_score()
        
        # 计算每个路径的综合评分
        scored_paths = []
        for path in self.alternative_paths:
            # 基础置信度
            base_score = path['confidence']
            
            # 新鲜度加成（新证据权重更高）
            freshness_bonus = (1 - path['age_bars'] / self.config.PATH_MAX_AGE_BARS) * 0.2
            
            # 连续性惩罚（跳转距离越远，惩罚越大）
            if not path.get('direct_jump', False):
                jump_distance = self._calculate_state_distance(self.current_state, path['state'])
                continuity_penalty = jump_distance * 0.15  # 增加连续性惩罚权重
            else:
                continuity_penalty = 0
            
            # 状态切换方向惩罚（相反方向切换代价更高）
            direction_penalty = self._calculate_direction_penalty(self.current_state, path['state'])
            
            total_score = base_score + freshness_bonus - continuity_penalty - direction_penalty
            scored_paths.append({
                'state': path['state'],
                'score': total_score,
                'base_confidence': base_score,
                'freshness': freshness_bonus,
                'penalties': continuity_penalty + direction_penalty
            })
        
        # 找到最佳路径
        best_path = max(scored_paths, key=lambda x: x['score'])
        best_state = best_path['state']
        best_score = best_path['score']
        
        # 应用滞后性阈值：新路径必须显著优于当前状态
        hysteresis_threshold = self.config.STATE_SWITCH_HYSTERESIS  # 默认0.15（15%）
        
        # 如果最佳路径就是当前状态，直接返回（无需切换）
        if best_state == self.current_state:
            return best_state
        
        # 计算相对优势
        relative_advantage = (best_score - current_state_score) / max(current_state_score, 0.01)
        
        # 切换条件：
        # 1. 最佳分数超过选择阈值
        # 2. 相对优势超过滞后性阈值 或 当前状态分数低于最低阈值
        switch_condition = (
            best_score > self.config.PATH_SELECTION_THRESHOLD and
            (relative_advantage > hysteresis_threshold or 
             current_state_score < self.config.STATE_MIN_CONFIDENCE * 0.7)
        )
        
        if switch_condition:
            print(f"状态切换: {self.current_state} → {best_state}")
            print(f"  当前分数: {current_state_score:.3f}, 最佳分数: {best_score:.3f}")
            print(f"  相对优势: {relative_advantage:.1%}, 阈值: {hysteresis_threshold:.1%}")
            return best_state
        
        # 保持当前状态，但记录潜在的替代路径
        if relative_advantage > hysteresis_threshold * 0.5:
            print(f"状态保持（接近切换）: {self.current_state} ← {best_state}")
            print(f"  相对优势不足: {relative_advantage:.1%} < {hysteresis_threshold:.1%}")
        
        return None  # 不切换
    
    def _get_current_state_score(self):
        """
        获取当前状态的综合评分
        包括：状态置信度、新鲜度、遗产强度
        """
        if self.current_state == 'IDLE':
            return 0.0
        
        # 从状态字典获取置信度
        state_info = self.accumulation_states.get(self.current_state)
        if not state_info:
            return 0.0
        
        base_score = state_info.get('confidence', 0.0)
        
        # 新鲜度加成：最近更新的状态得分更高
        freshness = self._get_state_freshness(self.current_state)
        freshness_bonus = freshness * 0.1
        
        # 遗产强度加成
        heritage_bonus = state_info.get('heritage_score', 0.0) * 0.2
        
        return base_score + freshness_bonus + heritage_bonus
    
    def _calculate_direction_penalty(self, from_state, to_state):
        """
        计算方向切换惩罚
        吸筹←→派发的切换代价最高，同方向切换代价较低
        """
        # 状态方向分类
        accumulation_states = ['PS', 'SC', 'AR', 'ST', 'TEST', 'UTA', 'SPRING', 'SO', 'LPS', 'mSOS', 'MSOS', 'JOC', 'BU']
        distribution_states = ['PSY', 'BC', 'AR', 'ST', 'UT', 'UTAD', 'LPSY', 'mSOW', 'MSOW']
        
        from_is_accum = from_state in accumulation_states
        to_is_accum = to_state in accumulation_states
        from_is_dist = from_state in distribution_states
        to_is_dist = to_state in distribution_states
        
        # 吸筹←→派发的方向切换（代价最高）
        if (from_is_accum and to_is_dist) or (from_is_dist and to_is_accum):
            return 0.3  # 30%惩罚
        
        # 同方向内部切换（代价中等）
        if (from_is_accum and to_is_accum) or (from_is_dist and to_is_dist):
            return 0.1  # 10%惩罚
        
        # 其他情况（如IDLE到任何状态）
        return 0.05
```

#### 3.3.4 设计原则
1. **价格优先原则**：关键价格突破触发状态重置，防止路径依赖错误
2. **并行路径跟踪**：同时维护多个可能性，避免"非此即彼"的思维定式

4. **防死锁机制**：超时强制重置，确保状态机不会卡在无效状态

```python
class EvidenceChainManager:
    def __init__(self):
        self.chains = {}
        self.cross_state_evidence = {}
    
    def get_state_confidence(self, state):
        """
        计算状态的综合置信度（加权平均 + 时间衰减 + 数量加成）
        """
        if state not in self.chains:
            return 0.0
        
        evidences = self.chains[state]
        if not evidences:
            return 0.0
        
        total_weight = sum(e['weight'] for e in evidences)
        weighted_sum = sum(e['confidence'] * e['weight'] for e in evidences)
        base_confidence = weighted_sum / total_weight
        
        time_factor = self._calculate_time_factor(evidences)
        count_factor = min(len(evidences) / 5, 1.0) * 0.2
        
        return min(base_confidence * time_factor + count_factor, 1.0)
```

### 4.4 激进信号与确认信号分层机制

#### 4.4.1 信号分层设计原则（结构确认优先）
传统威科夫依赖"收盘确认"，但在实战中（特别是加密/高频市场），等待H4收盘会导致严重滞后。本系统采用"结构确认"替代"时间确认"：
1. **激进信号（Aggressive Signal）**：M15/M5级别的预判，基于威科夫微观结构，仓位轻（25%）
2. **确认信号（Confirmed Signal）**：M15级别有效站稳H4关键结构位（Creek/Ice）3根K线，用于加仓和验证
3. **信号升级机制**：激进信号在获得结构确认后升级为确认信号
4. **结构确认标准**：价格突破H4关键位 + M15站稳3根K线 + 成交量健康（突破放量/站稳缩量）

#### 4.4.2 信号分层实现
```python
class SignalHierarchyManager:
    def __init__(self, config):
        self.config = config
        self.active_signals = {
            'aggressive': [],  # 激进信号列表
            'confirmed': []    # 确认信号列表
        }
        self.signal_upgrade_history = []  # 信号升级历史
    
    def evaluate_signal(self, timeframe, signal_data, market_context):
        """
        评估信号并分层
        返回: {
            'type': 'AGGRESSIVE'/'CONFIRMED'/'REJECTED',
            'confidence': float,
            'recommended_action': dict,
            'upgrade_conditions': list
        }
        """
        # 1. 基础信号评估
        base_evaluation = self._evaluate_base_signal(signal_data, market_context)
        
        if not base_evaluation['valid']:
            return {'type': 'REJECTED', 'reason': base_evaluation['reason']}
        
        # 2. 根据时间框架分层
        if timeframe in ['M15', 'M5']:
            # 小周期 → 激进信号
            signal_type = 'AGGRESSIVE'
            confidence = base_evaluation['confidence'] * 0.7  # 小周期信号置信度折扣
            position_size = self.config.AGGRESSIVE_POSITION_SIZE  # 默认25%
            
            # 设置升级条件
            upgrade_conditions = self._define_upgrade_conditions(timeframe, signal_data)
            
        elif timeframe in ['H4', 'D1']:
            # 大周期 → 确认信号
            signal_type = 'CONFIRMED'
            confidence = base_evaluation['confidence']
            position_size = self.config.CONFIRMED_POSITION_SIZE  # 默认100%
            upgrade_conditions = []  # 确认信号无需升级
        
        else:
            signal_type = 'REJECTED'
            confidence = 0
        
        # 3. 检查激进信号是否需要等待确认
        if (signal_type == 'AGGRESSIVE' and 
            self._requires_confirmation(signal_data, market_context)):
            
            return {
                'type': 'AGGRESSIVE_PENDING',
                'confidence': confidence,
                'recommended_action': {
                    'action': 'WAIT_FOR_CONFIRMATION',
                    'required_confirmation': 'STRUCTURE_BREAK',  # 等待结构确认（M15站稳H4关键位3根K线）
                    'max_wait_bars': 3  # 最多等待3根M15 K线验证结构站稳
                },
                'upgrade_conditions': upgrade_conditions
            }
        
        return {
            'type': signal_type,
            'confidence': confidence,
            'recommended_action': {
                'action': 'ENTER' if signal_type in ['AGGRESSIVE', 'CONFIRMED'] else 'WAIT',
                'position_size': position_size,
                'stop_loss': self._calculate_stop_loss(signal_data, signal_type),
                'take_profit': self._calculate_take_profit(signal_data, signal_type)
            },
            'upgrade_conditions': upgrade_conditions
        }
    
    def process_confirmation(self, aggressive_signal, confirmation_data):
        """
        处理信号确认，实现信号升级
        """
        # 检查升级条件是否满足
        upgrade_possible = self._check_upgrade_conditions(
            aggressive_signal, 
            confirmation_data
        )
        
        if upgrade_possible:
            # 执行信号升级
            upgraded_signal = self._upgrade_signal(aggressive_signal, confirmation_data)
            
            self.signal_upgrade_history.append({
                'aggressive_signal': aggressive_signal,
                'confirmation_data': confirmation_data,
                'upgraded_signal': upgraded_signal,
                'timestamp': get_current_time()
            })
            
            # 从激进信号列表移除，添加到确认信号列表
            self.active_signals['aggressive'].remove(aggressive_signal)
            self.active_signals['confirmed'].append(upgraded_signal)
            
            return {
                'upgraded': True,
                'new_signal': upgraded_signal,
                'recommended_action': {
                    'action': 'ADD_POSITION',  # 加仓
                    'position_size': self.config.CONFIRMED_ADDITION_SIZE,  # 默认50%
                    'reason': 'SIGNAL_CONFIRMED'
                }
            }
        
        return {'upgraded': False}
    
    def _define_upgrade_conditions(self, timeframe, signal_data):
        """
        定义激进信号的升级条件（结构确认替代时间确认）
        核心：M15级别有效站稳H4关键结构位（Creek/Ice）N根K线即视为确认
        """
        conditions = []
        
        if timeframe == 'M15':
            # M15激进信号 → 需要H4结构确认（非时间确认）
            conditions.append({
                'type': 'STRUCTURE_CONFIRMATION',
                'required': 'H4_KEY_LEVEL_BREAK',
                'condition': lambda market_data: (
                    # 条件1：价格突破H4关键结构位（Creek阻力或Ice支撑）
                    self._is_h4_structure_break(signal_data, market_data) and
                    # 条件2：在M15级别站稳该结构位至少3根K线
                    self._is_m15_holding_level(signal_data, market_data, min_bars=3) and
                    # 条件3：成交量健康（突破放量，站稳缩量）
                    self._has_healthy_volume_profile(signal_data, market_data)
                ),
                'description': 'M15有效突破并站稳H4关键结构位（Creek/Ice）'
            })
            

        
        elif timeframe == 'M5':
            # M5激进信号 → 需要M15结构确认 + H4结构锚定
            conditions.append({
                'type': 'PATTERN_CONFIRMATION',
                'required': 'M15_STRUCTURE',
                'condition': lambda data: (
                    # M15出现威科夫结构（不一定是完整形态）
                    self._has_wyckoff_structure(data, min_confidence=0.6) and
                    # 结构位置与H4关键位对齐
                    self._is_aligned_with_h4_structure(data, signal_data)
                ),
                'description': 'M15出现威科夫结构且与H4关键位对齐'
            })
            conditions.append({
                'type': 'STRUCTURE_CONFIRMATION',
                'required': 'H4_STRUCTURE_ANCHOR',
                'condition': lambda market_data: (
                    # H4有明确的结构位（Creek/Ice）作为锚定
                    market_data.get('h4_has_clear_structure', False) and
                    # M5入场点在该结构位附近（±2%）
                    abs(signal_data['entry_price'] - market_data['h4_structure_level']) / 
                    market_data['h4_structure_level'] < 0.02
                ),
                'description': 'H4有明确结构位作为交易锚点'
            })
        
        return conditions
    
    def _requires_confirmation(self, signal_data, market_context):
        """
        判断激进信号是否需要等待确认
        规则：大周期处于关键位置或冲突状态时需要结构确认（非时间确认）
        """
        # 规则1：日线处于派发区间 → 需要H4结构确认
        if (market_context.d1_phase == 'DISTRIBUTION' and
            signal_data['direction'] == 'LONG'):
            return True
        
        # 规则2：价格接近关键阻力/支撑 → 需要结构确认
        if (abs(signal_data['entry_price'] - market_context.key_resistance) / 
            market_context.key_resistance < 0.02):
            return True
        
        # 规则3：成交量未达标 → 需要确认
        if signal_data['volume_ratio'] < 1.5:
            return True
        
        # 规则4：H4缺乏明确结构位（Creek/Ice不清晰）→ 需要更严格确认
        if not market_context.get('h4_has_clear_structure', True):
            return True
        
        # 规则5：多周期冲突（如D1派发 vs H4吸筹）→ 需要结构确认
        if market_context.get('multi_timeframe_conflict', False):
            return True
        
        return False
```

### 5.1 WFA回测引擎与防过拟合机制

#### 5.1.1 Walk-Forward Analysis (WFA) 架构
```python
class WFABacktestEngine:
    def __init__(self, historical_data, initial_weights, config):
        self.historical_data = historical_data
        self.weights = initial_weights
        self.config = config
        self.wfa_windows = []  # WFA时间窗口
        self.performance_metrics = {}
        self.overfitting_scores = {}  # 过拟合评分
        
    def run_wfa_analysis(self):
        """
        执行Walk-Forward Analysis，防止过拟合
        返回: {
            'is_overfitting': bool,
            'out_of_sample_performance': dict,
            'stability_score': float,
            'recommended_adjustments': dict
        }
        """
        # 1. 划分训练集和测试集（滚动窗口）
        windows = self._create_wfa_windows()
        
        all_metrics = []
        weight_evolution = []
        
        for i, window in enumerate(windows):
            train_data = window['train']
            test_data = window['test']
            
            # 2. 在训练集上优化权重
            optimized_weights = self._optimize_on_train_set(train_data)
            weight_evolution.append(optimized_weights)
            
            # 3. 在测试集上验证性能
            test_metrics = self._evaluate_on_test_set(test_data, optimized_weights)
            all_metrics.append(test_metrics)
            
            # 4. 记录窗口性能
            self.wfa_windows.append({
                'window_id': i,
                'train_period': window['train_period'],
                'test_period': window['test_period'],
                'optimized_weights': optimized_weights,
                'test_metrics': test_metrics
            })
        
        # 5. 计算过拟合指标
        overfitting_analysis = self._analyze_overfitting(all_metrics, weight_evolution)
        
        # 6. 决定是否采纳新权重
        adoption_decision = self._make_adoption_decision(overfitting_analysis)
        
        return {
            **overfitting_analysis,
            'adoption_recommended': adoption_decision['recommended'],
            'adjusted_weights': adoption_decision.get('adjusted_weights', self.weights),
            'adjustment_reason': adoption_decision.get('reason', 'N/A')
        }
    
    def _create_wfa_windows(self):
        """
        创建WFA滚动窗口
        示例：训练集60天，测试集20天，滚动步长10天
        """
        windows = []
        total_bars = len(self.historical_data)
        
        train_size = self.config.WFA_TRAIN_DAYS  # 默认60天
        test_size = self.config.WFA_TEST_DAYS    # 默认20天
        step_size = self.config.WFA_STEP_DAYS    # 默认10天
        
        start_idx = 0
        while start_idx + train_size + test_size <= total_bars:
            train_end = start_idx + train_size
            test_end = train_end + test_size
            
            windows.append({
                'train': self.historical_data[start_idx:train_end],
                'test': self.historical_data[train_end:test_end],
                'train_period': (start_idx, train_end),
                'test_period': (train_end, test_end)
            })
            
            start_idx += step_size
        
        return windows
    
    def _optimize_on_train_set(self, train_data, max_iterations=50):
        """
        在训练集上优化权重，包含调整限制
        """
        current_weights = self.weights.copy()
        best_weights = current_weights.copy()
        best_performance = self._evaluate_performance(train_data, current_weights)
        
        for iteration in range(max_iterations):
            # 生成变异权重（有限制）
            mutated_weights = self._mutate_weights_with_limits(
                current_weights, 
                self.config.MAX_ADJUSTMENT_PERCENT
            )
            
            # 评估变异性能
            mutated_performance = self._evaluate_performance(train_data, mutated_weights)
            
            # 检查是否过拟合（训练集性能提升但可能过拟合）
            if self._is_significant_improvement(mutated_performance, best_performance):
                # 进一步检查稳定性
                stability_check = self._check_weight_stability(
                    best_weights, 
                    mutated_weights,
                    train_data
                )
                
                if stability_check['stable']:
                    best_weights = mutated_weights.copy()
                    best_performance = mutated_performance
            
            # 防止过度优化
            if iteration > 10 and not self._is_improving(best_performance, iteration):
                break
        
        return best_weights
    
    def _mutate_weights_with_limits(self, weights, max_adjustment_percent):
        """
        带限制的权重变异（每次调整不超过N%）
        """
        mutated = weights.copy()
        
        for key in self.config.EVOLUTION_PARAMS:
            if key in mutated:
                current_value = mutated[key]
                
                # 计算最大调整幅度
                max_change = abs(current_value) * (max_adjustment_percent / 100)
                
                # 生成随机调整（在限制范围内）
                adjustment = random.uniform(-max_change, max_change)
                
                # 应用调整
                mutated[key] = current_value + adjustment
                
                # 确保值在合理范围内
                if key.endswith('_THRESHOLD') or key.endswith('_CONFIDENCE'):
                    mutated[key] = max(0.0, min(1.0, mutated[key]))
                elif key.endswith('_MULTIPLIER'):
                    mutated[key] = max(0.1, min(10.0, mutated[key]))
        
        return mutated
    
    def _analyze_overfitting(self, all_metrics, weight_evolution):
        """
        分析过拟合情况
        """
        # 1. 性能稳定性分析
        performance_std = np.std([m['sharpe_ratio'] for m in all_metrics])
        performance_mean = np.mean([m['sharpe_ratio'] for m in all_metrics])
        
        # 2. 权重稳定性分析
        weight_stability = self._calculate_weight_stability(weight_evolution)
        
        # 3. 训练-测试差异分析
        train_test_gaps = []
        for window in self.wfa_windows:
            # 估算训练集性能（简化）
            train_perf = self._estimate_train_performance(window)
            test_perf = window['test_metrics']['sharpe_ratio']
            gap = abs(train_perf - test_perf)
            train_test_gaps.append(gap)
        
        avg_gap = np.mean(train_test_gaps) if train_test_gaps else 0
        
        # 4. 综合过拟合评分
        overfitting_score = (
            performance_std * 0.3 +
            (1 - weight_stability) * 0.3 +
            avg_gap * 0.4
        )
        
        is_overfitting = overfitting_score > self.config.OVERFITTING_THRESHOLD
        
        return {
            'is_overfitting': is_overfitting,
            'overfitting_score': overfitting_score,
            'performance_stability': 1 - performance_std,
            'weight_stability': weight_stability,
            'train_test_gap': avg_gap,
            'out_of_sample_performance': {
                'mean_sharpe': performance_mean,
                'min_sharpe': min(m['sharpe_ratio'] for m in all_metrics),
                'max_sharpe': max(m['sharpe_ratio'] for m in all_metrics),
                'win_rate_mean': np.mean([m['win_rate'] for m in all_metrics])
            }
        }
    
    def _make_adoption_decision(self, overfitting_analysis):
        """
        决定是否采纳新权重
        """
        if overfitting_analysis['is_overfitting']:
            return {
                'recommended': False,
                'reason': f"过拟合评分过高: {overfitting_analysis['overfitting_score']:.3f}",
                'suggestion': '减少调整幅度或增加WFA测试窗口'
            }
        
        # 检查性能提升是否显著
        current_perf = self._evaluate_current_performance()
        new_perf = overfitting_analysis['out_of_sample_performance']['mean_sharpe']
        
        improvement = (new_perf - current_perf['sharpe_ratio']) / abs(current_perf['sharpe_ratio']) * 100
        
        if improvement >= self.config.MIN_IMPROVEMENT_PERCENT:
            # 计算平滑过渡的权重（新旧权重混合）
            alpha = self.config.WEIGHT_SMOOTHING_FACTOR  # 默认0.3
            adjusted_weights = {}
            for key in self.weights:
                adjusted_weights[key] = (
                    alpha * self.weights[key] + 
                    (1 - alpha) * self._get_average_weight(key)
                )
            
            return {
                'recommended': True,
                'improvement_percent': improvement,
                'adjusted_weights': adjusted_weights,
                'reason': f"性能提升{improvement:.1f}%，且无过拟合"
            }
        else:
            return {
                'recommended': False,
                'reason': f"性能提升不足: {improvement:.1f}% < {self.config.MIN_IMPROVEMENT_PERCENT}%"
            }
```

### 7.2 C阶段决定性作用：趋势转换的唯一确认点

**问题：** "盘整区就代表着为下一阶段的蓄力，他可能是吸筹，派发，再吸筹，再派发，但是都是在C阶段才能确定。所以我们在C阶段出现，才是我们确定下一段趋势的逻辑。"

**回答：** 完全正确。本系统核心设计哲学就是**"只在C阶段（盘整区）分析，只在突破确认后行动"**：

#### 1. C阶段的决定性意义（威科夫核心）
- **蓄力阶段**：C阶段是主力资金建仓/派发的"蓄力区"，所有准备工作在此完成
- **方向未定**：在C阶段内部，价格可能多次测试边界，呈现"吸筹-派发-再吸筹"的复杂结构
- **唯一确认点**：只有C阶段完成（JOC突破或突破失败）才能确定下一阶段趋势方向

#### 2. 系统设计中的C阶段体现
- **TR锁定机制**：一旦识别高质量TR（C阶段），系统锁定该区间，专注分析内部结构
- **状态机专注C阶段**：所有13个吸筹节点和9个派发节点都只在TR内部有效
- **突破确认原则**：价格有效突破TR边界（结构确认）前，不做出趋势方向判断
- **"蓄力-突破"逻辑链**：
  ```
  TR识别 → 内部结构分析（吸筹/派发证据收集） → 边界测试 → 突破确认 → 趋势跟随
  ```

#### 3. 实战优势（避免趋势中段的误判）
- **不预测趋势**：系统不试图预测趋势开始或结束，只识别C阶段和突破确认
- **高胜率位置**：只在"蓄力完成-突破确认"的高胜率位置入场
- **明确止损**：突破失败（重回TR）即止损，逻辑清晰

#### 4. 多周期C阶段嵌套
- **周线C阶段**：决定数月趋势方向
- **日线C阶段**：决定数周趋势方向  
- **4小时C阶段**：决定数日趋势方向
- **15分钟C阶段**：提供精确入场点

**结论：** 系统严格遵循"只在C阶段分析，只在突破确认后行动"的威科夫核心原则，避免在趋势中段猜测方向，确保每次交易都建立在"蓄力完成-突破确认"的高概率基础之上。

## 8. 新要求深度问题解决方案

### 8.1 问题清单与解决方案概要

基于策略架构师的最新深度分析，以下8个隐形问题已制定解决方案：

#### 问题1：遗产数学黑盒陷阱
- **风险**：遗产系数硬编码（如1.2、0.8），误差逐级放大，形成不可调试的黑盒
- **解决方案**：
  - 遗产系数参数化（JSON配置文件，所有lambda系数外部可调）
  - 实时可视化调试面板（HeritageVisualizer模块）
  - 详细日志记录遗产分数来源和传递路径

#### 问题2：市场体制循环论证
- **风险**：Regime判断依赖K线形态，K线形态识别又依赖Regime，形成逻辑死锁
- **解决方案**：
  - 无状态启动模式（RegimeDetector独立模块）
  - 初期仅用ATR、ADX、历史波动率判断市场体制
  - 状态机确立明确TR/趋势后才介入Regime因子

#### 问题3：三根K线伪确认
- **风险**：简单计数器易被SFP（摆动失败模式）欺骗，价格横盘3-5根后跌回区间
- **解决方案**：
  - "突破+回踩不破"逻辑替代简单横盘计数
  - BreakoutValidator模块检测反向测试K线（缩量、下影线触及支撑）
  - "拒绝回调"（Refusal to go back）胜率验证

#### 问题4：几何拟合代码缺失
- **风险**：线性回归无法识别圆弧底、收敛三角形等曲线边界，导致完美吸筹底部踏空
- **解决方案**：
  - Pivot Spline Fitting（高低点样条插值）
  - 多项式拟合最近5个Pivot High/Low点
  - 生成动态曲线包络（Curved Envelope）替代直线通道

#### 问题5：状态爆炸与逻辑剪枝
- **风险**：40+术语 × 4周期 = 成千上万可能路径，导致决策瘫痪
- **解决方案**：
  - 概率剪枝机制（LogicPruner模块）
  - 置信度<30%路径立即丢弃
  - 仅保留Top 3逻辑链进行深度分析

#### 问题6：进化逻辑退化风险
- **风险**：AI为追求夏普比率可能阉割威科夫核心逻辑（如去掉Test环节）
- **解决方案**：
  - 逻辑基因保护区（EvolutionGuard模块）
  - VSA核心公式（Effort vs Result）、遗产传递规则禁止进化修改
  - 威科夫辩证灵魂保护机制

#### 问题7：跨周期时间差陷阱
- **风险**：大周期看多时，小周期可能已完成完整回调，僵化执行导致反复止损
- **解决方案**：
  - 节奏对齐（RhythmSynchronizer模块）
  - 大周期决定"能做什么"（战略方向）
  - 小周期结构决定"何时不能做"（战术时机）

#### 问题8：异常数据语义误判
- **风险**：无法区分机构造成的异常（真实信号）vs交易所宕机/假数据异常
- **解决方案**：
  - 多源互证（AnomalyValidator模块）
  - 交叉比对同板块品种相关性（如BTC/ETH）
  - 确认是否为全市场行为

### 8.2 第一阶段实施优先级调整

根据新要求，第一阶段（基础框架）增加两个关键任务：

1. **Regime独立模块开发**（最高优先级）
   - 目标：打破循环依赖，提供稳定的市场体制判断
   - 交付：`market_regime.py`（仅基于ATR/ADX/波动率）

2. **可视化调试面板**（最高优先级）
   - 目标：消除黑盒，实现逻辑透明
   - 交付：`heritage_panel.py` + `threshold_monitor.py`（Plotly实时面板）

3. **系统精简原则**
   - 删除非必要功能，避免臃肿
   - 聚焦核心：识别趋势vs盘整，盘整区间高空低多，趋势转换后单边交易
   - 保持自动进化能力，但不增加复杂性
